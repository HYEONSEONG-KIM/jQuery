:contains(text) -> 지정한 텍스트를 포함하는 요소
:has(selector) -> 대상요소 중 자식요소도 없고 텍스트도 없는 요소
:empty -> 대상요소 중 자식요소도 없고 텍스트도 없는 요소
:parent -> 대상요소 중 자식요소를 갖거나 텍스트를 갖고 있는 요소와 일치


//jQuery가 포함된 span태그 의 테두리 파랑 :
$('span:contains("jQuery")').css('border', '2px solid blue');

//jQuery가 포함되지 않은 span 태그 의 테두리 빨강 :
$('span').not('span:contains("jQuery")').css('border', '2px solid red');
$('span:not(:contains(jQuery))').css('border', '2px solid red');

//span을 갖는 div요소 태그 의 테두리 그린
$('div:has(span)').css('border','1px solid green');

//아무것도 없는 빈 div 태그 의 테두리
$('div:empty').css('border','1px solid yellow');

//div 후손 span 태그 의 글자색 빨강, 크기,굵게
$('#box1 div span').css({'color':'red', 'font-size' : '2.0em', 'font-weigth' : 'bold'});
